no_cores <- detectCores()
registerDoParallel(cores = no_cores)
cl <- makeCluster(no_cores, type="FORK")

superSum <- function(opsum) {
  y <- solve_LSAP(x, maximum = TRUE)
  sum(x[cbind(seq_along(y), y)]) 
}
print(superSum(opsum))
for (i in 1:1000) {
    ptime <- system.time(
        {
          parLapply(cl, 1, superSum)
        })
    parTime<-c(ptime)
}

minP<-min(parTime)
maxP<-max(parTime)
mean<-mean(parTime)
medianP<-median(parTime)
cat("Mean ", meanP, "Median:" , medianP, "Min:", minP, "Max:", maxP)
parLvector<-c(meanP, medianP, minP, maxP)
print(parLvector)
plot(parLvector, main = "parLApply", xlab =Mean     Median     Min     Max", ylab= "time")
