library(MASS)
r = range(faithful$waiting)
cat("\nRange: ")
print(r)
bins <- seq(43,96,by=6)
intervals <- cut(faithful$waiting, bins, right=FALSE)
print("Frequency distribution is:" )
print(cbind(table(intervals)))
sort(table(intervals), decreasing = TRUE)
mean = mean(table(intervals)) 
median =  median(table(intervals)) 
cat("\nMean is: ",mean)
cat("\nMedian is: ",median)
plot(table(intervals), type = “h”, main = “Erruption Histogram”, xlab = “Waiting Time”, ylab = “Erruptions”)
cat("\nIQR: ",quantile(table(intervals)))
boxplot(faithful$waiting)
cat("\nMost waiting time: ")
print(sort(table(intervals), decreasing = TRUE)[1])
